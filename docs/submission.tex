\section{Executive Summary}

\textbf{Meridian} is a production-grade performance management platform that replaces spreadsheet chaos, email bottlenecks, and offline review cycles with a structured, digital goal-setting and tracking portal.

\vspace{8pt}

\begin{graybox}
\textbf{Key Differentiators}
\vspace{4pt}
\begin{itemize}[leftmargin=1.5em, itemsep=3pt]
    \item \textbf{Zero-infrastructure demo} — fully functional without external services (\$0/month)
    \item \textbf{Production-ready architecture} — Prisma schema, middleware, RBAC, idempotency
    \item \textbf{All 4 UoM calculation types} — Min, Max, Timeline, Zero-based with mathematical correctness
    \item \textbf{Enterprise integrations} — Microsoft Entra ID, Graph API, Teams Adaptive Cards
    \item \textbf{8-panel analytics} — interactive Recharts visualizations with department roll-ups
\end{itemize}
\end{graybox}

\vspace{12pt}

\subsection{Key Design Decisions}

\begin{enumerate}[leftmargin=1.5em, itemsep=5pt]
    \item \textbf{Microservice-ready monolith} — services are logically isolated, ready for extraction at scale
    \item \textbf{Dual-mode data layer} — Zustand for \$0 demo; Prisma + PostgreSQL for production (zero code rewrite)
    \item \textbf{Idempotent operations} — \texttt{getOrCreateSheet()} follows the Stripe pattern; double-clicks are safe
    \item \textbf{Client-side analytics} — Zustand selectors compute on-the-fly; no separate analytics DB needed
\end{enumerate}

\section{Security \& Production Hardening}

\vspace{6pt}
\begin{itemize}[leftmargin=1.5em, itemsep=4pt]
    \item \textbf{Middleware} — Security headers, request IDs, auth guards on every route
    \item \textbf{Idempotency} — \texttt{ApiIdempotencyKey} prevents double-submits on critical operations
    \item \textbf{Outbox Pattern} — \texttt{OutboxEvent} decouples delivery from user transactions
    \item \textbf{Optimistic Locking} — \texttt{version} columns prevent write conflicts
    \item \textbf{Input Sanitization} — Zod schemas + HTML sanitization at every boundary
    \item \textbf{RBAC} — Role-based access at middleware, API, and UI layers
\end{itemize}
